/* -----------------> USANDO OPERADORES LOGICOS <----------------- */

/* Consulta por posto, estado,municipio, considerando produto,grupo do produto E nota = 5 */

SELECT POSTO.RAZAO_SOCIAL,AVALIACAO.NOTA, ENDERECO.UF, ENDERECO.MUNICIPIO, PRODUTO.GRUPO_DO_PRODUTO, PRODUTO.PRODUTO
FROM POSTO
INNER JOIN AVALIACAO
ON  POSTO.ID = AVALIACAO.ID_POSTO
INNER JOIN ENDERECO
ON POSTO.ID = ENDERECO.ID_POSTO
INNER JOIN PRODUTO
ON POSTO.ID = PRODUTO.ID_POSTO
WHERE ENDERECO.UF= 'SP' 
AND ENDERECO.MUNICIPIO = 'SAO PAULO'
AND PRODUTO.GRUPO_DO_PRODUTO = 'Gasolina'
AND PRODUTO.PRODUTO = 'GASOLINA C COMUM'
AND AVALIACAO.NOTA = 5
